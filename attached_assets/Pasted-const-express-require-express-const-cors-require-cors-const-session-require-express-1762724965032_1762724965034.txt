const express = require('express');
const cors = require('cors');
const session = require('express-session');
const swaggerUi = require('swagger-ui-express');
const swaggerSpec = require('./config/swagger');
const connectDB = require('./config/database');
const passport = require('./config/passport');
const authRoutes = require('./routes/auth');
const itemsRoutes = require('./routes/items');
const notificationsRoutes = require('./routes/notifications');
const userRoutes = require('./routes/users');
const adminRoutes = require('./routes/admin');
const chatRoutes = require('./routes/chat');
const recommendationsRoutes = require('./routes/recommendations');
const itemRequestsRoutes = require('./routes/itemRequests');
const contactRoutes = require('./routes/contact');
const paymentRoutes = require('./routes/payment');
const addressRoutes = require('./routes/address');
const ratingsRoutes = require('./routes/ratings');
const gamificationRoutes = require('./routes/gamification');
const { startNotificationServices } = require('./services/notificationService');
const { startExpirationCleanup } = require('./services/cronService');

const app = express();
const PORT = process.env.PORT || 3000;

// Connect to MongoDB
connectDB();

// Start notification services (cron jobs for expiry alerts and nearby items)
startNotificationServices();

// Start expiration cleanup cron job
startExpirationCleanup();

// CORS configuration - Allow requests from Vercel frontend
const allowedOrigins = [
  'https://grocery-share-frontend.vercel.app',
  'http://localhost:5000',
  'http://localhost:3000',
  'http://localhost:5173'
];

// Add custom frontend URLs from environment if provided
if (process.env.FRONTEND_URL) {
  allowedOrigins.push(process.env.FRONTEND_URL);
}

app.use(cors({
  origin: function (origin, callback) {
    // Allow requests with no origin (mobile apps, Postman, curl, etc.)
    if (!origin) return callback(null, true);
    
    if (allowedOrigins.indexOf(origin) !== -1) {
      callback(null, true);
    } else {
      console.log('âŒ CORS blocked request from:', origin);
      callback(new Error('Not allowed by CORS'));
    }
  },
  credentials: true,
  methods: ['GET', 'POST', 'PUT', 'DELETE', 'OPTIONS'],
  allowedHeaders: ['Content-Type', 'Authorization']
}));

// Use JSON middleware
app.use(express.json());

// Session configuration for passport
app.use(session({
  secret: process.env.SESSION_SECRET || 'default-session-secret',
  resave: false,
  saveUninitialized: false,
  cookie: {
    secure: process.env.NODE_ENV === 'production',
    maxAge: 24 * 60 * 60 * 1000
  }
}));

// Initialize passport
app.use(passport.initialize());
app.use(passport.session());

// Swagger API Documentation
app.use('/docs', swaggerUi.serve, swaggerUi.setup(swaggerSpec, {
  customCss: '.swagger-ui .topbar { display: none }',
  customSiteTitle: 'Grocery Share API Docs'
}));

// Health check
app.get('/', (req, res) => {
  res.send('Welcome to Grocery Share Backend API! Visit /docs for API documentation.');
});

// API health check endpoint
app.get('/api/health', (req, res) => {
  res.json({ message: 'API is working' });
});

// Mount authentication routes
app.use('/auth', authRoutes);

// Mount items routes
app.use('/items', itemsRoutes);

// Mount notifications routes
app.use('/notifications', notificationsRoutes);

// Mount users routes
app.use('/users', userRoutes);

// Mount admin routes
app.use('/admin', adminRoutes);

// Mount chat routes
app.use('/chat', chatRoutes);

// Mount recommendations routes
app.use('/recommendations', recommendationsRoutes);

// Mount item requests routes
app.use('/item-requests', itemRequestsRoutes);

// Mount contact routes
app.use('/contact', contactRoutes);

// Mount payment routes
app.use('/payment', paymentRoutes);

// Mount address routes
app.use('/address', addressRoutes);

// Mount ratings routes
app.use('/ratings', ratingsRoutes);

// Mount gamification routes
app.use('/gamification', gamificationRoutes);

// Error handling middleware
app.use((err, req, res, next) => {
  // Log error details properly
  console.error('âŒ Error occurred:');
  console.error('Message:', err.message);
  console.error('Stack:', err.stack);
  
  // Handle Multer errors
  if (err.name === 'MulterError') {
    if (err.code === 'LIMIT_FILE_SIZE') {
      return res.status(413).json({ 
        error: 'File too large. Maximum size is 5MB per file.',
        details: err.message 
      });
    }
    if (err.code === 'LIMIT_FILE_COUNT') {
      return res.status(400).json({ 
        error: 'Too many files. Maximum is 5 images.',
        details: err.message 
      });
    }
    if (err.code === 'LIMIT_UNEXPECTED_FILE') {
      return res.status(400).json({ 
        error: 'Unexpected file field. Use "images" field name.',
        details: err.message 
      });
    }
    return res.status(400).json({ 
      error: 'File upload error',
      details: err.message 
    });
  }
  
  // Handle Cloudinary errors
  if (err.message && err.message.includes('cloudinary')) {
    console.error('ğŸ”´ Cloudinary error detected');
    return res.status(500).json({ 
      error: 'Image upload service error. Please try again later.',
      details: process.env.NODE_ENV === 'development' ? err.message : 'Cloud storage error'
    });
  }
  
  // Handle general errors
  res.status(err.status || 500).json({ 
    error: err.message || 'Internal server error',
    details: process.env.NODE_ENV === 'development' ? err.stack : undefined
  });
});

// Start server - bind to 0.0.0.0 for Render deployment
app.listen(PORT, '0.0.0.0', () => {
  console.log(`âœ… Server running on port ${PORT}`);
  console.log(`ğŸŒ Environment: ${process.env.NODE_ENV || 'development'}`);
  console.log(`ğŸ”’ CORS enabled for: ${allowedOrigins.join(', ')}`);
  console.log(`ğŸ“¡ Ready to receive requests!`);
});
